               xmlns:s="library://ns.adobe.com/flex/spark" 
               xmlns:mx="library://ns.adobe.com/flex/mx" minWidth="955" minHeight="600" width="834" height="498" viewSourceURL="srcview/index.html">
    <fx:Declarations>
        <!-- 将非可视元素（例如服务、值对象）放在此处 -->
    </fx:Declarations>
    <fx:Script>
        <![CDATA[
            import format.swf.Cpool_info;
            import format.swf.DoABC;
            import format.swf.String_info;
            import format.swf.SwfFile;
            
            private var file:FileReference;
            private var data:ByteArray;
            private var swf:SwfFile;
            [Bindable]private var strings:Array;
            
            private function doSwf():void
            {
                swf = new SwfFile(data);//读取swf
                var doAbcs:Array = swf.tagReader.getTagReaderByType(82);//得到Tag82的reader
                var doAbc1:DoABC = doAbcs[0];//取第一个reader
                var constant_pool:Cpool_info = doAbc1.abcfile.constant_pool;//取得常量池
                var length:uint = constant_pool.string_count;
                strings = new Array;
                for (var i:int = 0; i < length; i++)
                {
                    var element:String_info = constant_pool.string[i] as String_info;
                    if(element)
                        strings.push({index:i,string:element.utf8str});
                    else
                        strings.push({index:i,string:0});
                }
            }
            protected function button1_clickHandler(event:MouseEvent):void
            {
                file = new FileReference();
                file.addEventListener(Event.SELECT,onSelect);
                file.addEventListener(Event.COMPLETE,onComplete);
                file.browse([new FileFilter("swf文件","*.swf")]);
            }
            
            private function onComplete(event:Event):void
            {
                data = file.data;
                doSwf();
            }
            private function onSelect(event:Event):void
            {
                file.load();
            }
        ]]>
    </fx:Script>
    <s:Button x="363" y="467" label="载入" click="button1_clickHandler(event)"/>
    <mx:DataGrid x="22" y="50" width="802" height="398" dataProvider="{strings}">
        <mx:columns>
            <mx:DataGridColumn headerText="String_pool" dataField="index"/>
            <mx:DataGridColumn headerText="值" dataField="string"/>
        </mx:columns>
    </mx:DataGrid>
    <s:Label x="10" y="10" text="String Reader demo" width="164" height="18"/>
</s:Application>
